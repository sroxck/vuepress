# TS基础-接口

接口可以为变量或函数指定需要实现的类型

# 通过下面的例子来看接口的工作流程

# 不使用接口

函数有一个参数,参数类型是对象(这里是用的字面量形式) 对象有一个size属性 类型是string

function printLabel(printObj:{size:string}){
    console.log(printObj.size);
}
printLabel({size:"333"})

# 使用接口

// 定义接口
interface printValue {
    size: string
}
// 参数的类型是这个接口,那么参数就必须符合这个接口中定义的内容
function printLabeltwo(printObj:printValue){
    console.log(printObj.size);
}
printLabel({size:"333"})

TIP

类型检查器不会去验证属性的顺序只要相应的属性存在并且类型也是对的就可以。

# 类型别名

type names=string let namess:names = "33" type obj = {name:string}

类型别名是使用type声明一个变量,这个变量设置的类型可以被使用,类似于接口,但是接口只能实现对象,类型别名可以实现基础类型,如string,number,在ts中建议尽量使用接口

# 接口的继承

接口的继承使用extends,会继承父接口中定义的类型

interface faceOne {
    face1:string
}
interface head extends faceOne {
    mouth:string
}
let sroxck:head={mouth:"small",face1:"big"}

# 函数类型的接口

使用接口定义函数类型的格式:(参数名:参数类型):返回值类型

interface fnimpl {
    (name:string,num:number):string
}

# 索引签名(可索引类型)

接口也可以描述那些可以通过索引获取的值的类型,例:

interface indexInterFace{
    [index:number]:string
}
let arrlist:indexInterFace=['1','2']
// 这里接口限制了索引的类型是number,索引返回值是string,适用于数组等

interface indexInterFace1{
    [index:string]:string
    name:string
}
let arrlist1:indexInterFace1={name:'name'}
// 对象类型

interface NumberDictionary {
    [index: string]: number;
    length: number;    // 可以，length是number类型
    // name: string       // 错误，`name`的类型与索引类型返回值的类型不匹配
  }
// 设置了索引签名的返回值类型,其他值的类型必须相同

// 索引签名可以设置成只读
interface readonlydic{
    readonly [index:number]:any
}
let arrlistthree : readonlydic = [1,3,4]
// arrlistthree[0] = 2 // 只读的索引签名是不可以通过索引修改值的

# 只读属性

ts接口中可以定义某个属性只允许在创建的时候修改值(即初始化赋值),之后就不允许修改(类似于const)

interface onlyread{
    readonly size:string
}
let only : onlyread = {size:"only"}
 only.size = "333" // error 只读不可修改

//数组也拥有只读方式,onlyArray<E> 只读数组将所有可变的方法和属性去掉了
let arr : ReadonlyArray<number> = [1,2,3]
 arr.push(3) // error arr上不存在push方法,因为push是可变方法
let arr2: number[] = []
// arr2 = arr 把readonly赋值给普通数组也不行
// 但是可以使用 as(类型断言)的方式赋值
arr2  = arr as number[] // 成立
