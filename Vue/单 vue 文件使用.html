# vue文件可以由 3 个部分组成 template, script, style

# template

  1. 承载主体 html, 只能有一个根元素.
  2. 可以使用上下文,调用 script 中的属性和方法
<template>
  <root>
    <content>
      ...
    </content>
    <main>
      {{message}}
    </main>
  </root>
</template>

# script

  1. .vue 文件中核心的部分,通过 exprot default 导出一个对象
  2. 导出对象中 对象中使用 vue 提供的语法即可
  3. 在 script 中可以使用正常的 js 语法,通常可以在 exprot 之前 import...from... 来导入需要的组件
<script>
export default {
  // 文件的name
  name: "",
  // 这里可以接收父级组件传递的参数
  prop:{
  },
  // 在这里注册组件,通过展开运算符加载
  components:{
    ...组件 1,
    ...组件 2
  },
  methods: {
    //调用接口 获取初始化数据的方法 可以用来做组件的数据刷新
    load(){
    },
    ///////////////////////////////////////////////////////////接口区域////////////////////////////////////////////////////////////////
    // 接口
    method1() {
    },
    ///////////////////////////////////////////////////////////事件区域////////////////////////////////////////////////////////////////
    // 组件事件 组件库里封装好的事件或普通定义的事件(单击、双击、悬浮。。。)
    method3(){
    },
    ///////////////////////////////////////////////////////////功能区域////////////////////////////////////////////////////////////////
    // 封装的功能代码方法(重置表单，表格排序。。。)
    method4() {
    },
  },
  // 侦听器,监听属性变化,执行操作
  watch:{
  }
  data() { // 单文件vue 的数据中心
    return { // 通过 return 的方式返回一个对象,这样每次被调用返回的都是一个新的实例
      //接口数据区域
      attribute1:[],//属性1说明
      //表单数据区域
      attribute2:{//属性2说明
        attribute1:'',//子属性1说明
        attribute2:'',//子属性2说明
        attribute3:'',//子属性3说明
      },
      //操作数据区域
      attribute3:'',//属性3说明
      //静态资源引用区域
      attribute4:require('引入的路径(引入模块,一般为图片)'),//属性4说明
      //其他数据区域
      attribute5:'',//属性5说明
    }
  },
  //这个周期是在实例组件刚创建时执行 没有初始化数据，也没有挂载DOM,不常用
  beforeCreate() {
  },
  //这个周期是在数据data初始化完成后执行
  created() {//在这个周期里面适合进行请求
    this.load()//请求初始接口数据
  },
  //这个周期是在挂载DOM前执行 但是数据的双向绑定还是显示{{}}，这是因为Vue采用了Virtual DOM（虚拟Dom）技术,先占住了一个坑。
  beforeMount() {
  },
  //这个周期是在挂载DOM后执行 将挂载前占位的地方填入数据
  mounted() {//这个周期适合执行初始化需要操作DOM的方法。
  },
  //这个周期是在组件的数据发生变化前执行 页面数据还是原来的数据 谨慎使用，因为页面更新数据的时候都会触发，在这里操作数据很影响性能和容易死循环。
  beforeUpdate() {
  },
  //这个周期是在组件的数据发生变化后执行 页面的数据是更新完成的 谨慎使用，因为页面更新数据的时候都会触发，在这里操作数据很影响性能和容易死循环。
  updated() 
  },
  //这个周期是在组件销毁之前执行，beforeDestroy无法阻止路由跳转
  beforeDestroy() {//这个周期适合做一些路由离开的时候操作(清除计时器。。。)
  },
  //这个周期是在销毁组件完毕后执行
  destroyed() {
  },
}
</script>

# style

  1. 页面的 css 区域,需要声明 lang的值,scss,less,styl,css 等
  2. 不声明 scoped,在 style 区域内写的样式会造成全局命名污染
  3. 声明 scoped ,组件样式修改无效通常是权重不够,scss 可以通过/deep/ 深度选择器来提权,css 可以使用 >>>
<style lang="scss" scoped>
  $colors: (
  "primary": #db9e3f,
  "white": #fff,
  "info":rgba(34,213,156,1),
  "light": #f9f9f9,
  "grey": #999,
  "dark": #222,
  "black": #000,
  "chinaRed":red,
  "danger": rgb(192, 75, 75),
);
  @each $key,$val in $colors {
    .text-#{$key} {
      color:$val
    }
    .bg-#{$key} {
      color:$val
    }
  }
  .main {
    color: map-get($colors, 'dark')
  }
</style>
